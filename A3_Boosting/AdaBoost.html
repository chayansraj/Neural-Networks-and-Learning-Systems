def adaBoost(X,Y, num_classifiers):
    
    nfeatures, nsamples = X.shape # We extract features and samples from input data using numpy
    weights = 1.0/len(Y) # Intialize weights
    classifier = [] # List for keeping all weak classifiers
    errors = [] # Error of each classifier
    features = [] # Best feature chosen in each weak classifier
    thres_list = [] # Best threshold chosen for each weak classifier
    Polar_list = [] # Polarity for each threshold and weak classifier
    alpha_list =[] # Alpha list for confidence in each weak classifier
    
    # First loop for the number of classifiers
    for classifiers in range(num_classifiers):
        
        min_loss = float('inf') # setting max. error value for comparison
        
        # Second loop to go over every feature
        for feat in range(nfeatures):

            feature = X[feat,:] # Extracting every feature 
 
            thresholds = np.unique(feature) # setting thresholds as all the unique values of a sampled feature value
 
            # third loop to go over every threshold
            for thres in thresholds:
                p=1 # initial polarity
                preds=np.ones(nsamples) # initializing predictions for every thresholds 
                preds[feature<thres]=-1 # setting -1 for every smaller value of feature than threshold

                loss = WeakClassifierError(C = preds, D = weights, Y =Y) # calculating loss for a feature and corresponding threshold

                if loss > 0.5: 
                    p = -1
                    loss = 1-loss

                if loss < min_loss:
                    min_loss = loss
                    T = thres
                    P = p
                    feature_idx = feat
                    
        EPS = 1e-15 # to prevent underflow/overflow
        alpha_t = 0.5*np.log((1.0-min_loss+EPS)/(min_loss+EPS)) # calculating alpha as given in slides
        prediction = WeakClassifier(T , P , X = X[feature_idx,:]) # getting predictions for each weak classifier
        features.append(feature_idx) # storing the best feature for each weak classifier
        thres_list.append(T) # storing the best threshold value for best feature above for each weak classifier
        Polar_list.append(P) # storing the best polarity for above thresholds value for best feature above for each weak classifier
        alpha_list.append(alpha_t) # storing alpha for each weak classifier
        weights = weights * np.exp(-alpha_t * Y * prediction) # reinitializing weights for incorrectly classified samples
        weights = weights / np.sum(weights) # normalizing the weights
        classifier.append(alpha_t*prediction) # adding each weak classifier
        errors.append(sum((np.sign(np.sum(classifier, axis=0)) !=Y))/len(Y)) # adding error for each weak classifier
        

    return np.sign(np.sum(classifier, axis=0)), errors, features, thres_list, Polar_list, alpha_list

# Evaluate your strong classifier here
# Evaluate on both the training data and test data, but only the test accuracy
# can be used as a performance metric since the training accuracy is biased.

def predict(X,Y,num_classifier, final_alpha, final_thres, final_pol, final_feat):
    
    clf_preds = [final_alpha[i] * WeakClassifier(final_thres[i], final_pol[i], X[final_feat[i],:]) for i in range(num_classifier)]
    pred = np.sum(clf_preds,axis=0)
    pred = np.sign(pred)
    
    return pred, sum(pred == Y)/len(Y)
